//Решить задачу - Сумма каких-либо двух цифр заданного трехзначного натурального числа N равна третьей цифре.
#include <iostream>
#include <stdio.h>
#include <math.h>

using namespace std;
int main() 
{
    int x , y, z;
    cout << "Введите трехзначное число x = ";
    cin >> x;
    
    y = x % 10;
    x = x / 10;
    z = x % 10;
    x = x / 10;
    
    if (x + y == z)
        cout << "True";
    else
        if (z + y == x)
        cout << "True";
        else
            if (x + z == y)
            cout << "True";
            else
            cout << "False";

    return 0;
}
